Javascript Async Functions

1) Compare async and sync calling

    const resolveAfter2Seconds = () => {
        return new Promise((resolve) => {
            setTimeout(()=>{
                resolve('resolve');
            }, 2000);
        });
    }
    
    const asyncCall = async () => {
        console.log('async calling start ...');
        let result = await resolveAfter2Seconds();
        console.log(result);
        console.log('async calling end ...');
    }
    
    const syncCall = () => {
        console.log('sync calling start ...');
        let result = resolveAfter2Seconds();
        console.log(result);
        console.log('sync calling end ...');
    }
    
    syncCall();
    
    asyncCall();

Note: check the console to see the results.
